# 💻  Linux Command2
```
$ rm -rf 이름 : '이름'파일 삭제
$ pwd : 현재 위치 확인
$ wc -l 이름 : '이름'파일의 행 수

$ mv 이름1 이름2 : '이름1'파일을 '이름2'파일로 이동 (존재 시 덮어쓰기)
$ mv 이름1 ../이름2/ : '이름1'을 '이름2'디렉토리로 이동
$ cat > 이름 : '이름'파일 내용 작성후 Ctrl+D 로 저장
$ cp 이름1 이름2 : '이름1'파일을 '이름2'파일로 복사
$ cp 이름 dir/ : '이름'파일을 dir디렉토리 하위로 복사
$ cp -r dir1/ dir2/ : dir1을 dir2로 복사(하위에 있는 것 모두)

$ seq 0 20 100 : sequency생성, 0부터 시작, 20씩 증가, 100까지

$ head 이름 : '이름'파일내용의 10행만 출력
$ head -n5 이름 : '이름'파일내용의 5행만 출력
$ head 이름1 이름2 : '이름1'과 '이름2'파일내용을 10행까지 출력
$ head 이름1 이름2 -q : 위와 같지만 구분선 생략

$ tail 이름 : '이름'파일내용의 뒤에서부터 10행만 출력
$ tail -n5 이름 : '이름'파일내용 뒤에서부터 5행만 출력

$ split 이름 : '이름'파일 내용을 나눔 (임시적인 이름 적용 xaa, xab...)
$ split -d 이름 : '이름'파일내용을 나눈 파일의 이름을 정수로 변경 (x01, x02...)
$ split -l 2000 이름 : '이름'파일내용을 2000행씩 나눔
$ split -b 5000 이름 : '이름'파일내용을 5000byte씩 나눔 (내용 잘릴 가능성ㅇ)
$ split -c 5000 이름 : '이름'파일내용을 5000byte가 넘지 않게 나눔
$ split -d --additional-suffix=.txt 이름 backup_
 : '이름'파일을 나눈 파일의 이름을 직접 지정(backup_00.txt, backup_01.txt ...)

```
